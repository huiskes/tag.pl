# tag configuration file
# see manual.txt for more detailed help
# make sure all used directories exist 

###########################################################################
#
# MAIN SETTINGS
#
###########################################################################

# viewing mode
# use single or grid
# single: single image view; grid: image grid view 
MODE grid

# image directory
IM_DIR C:\Users\Mark\Databases\mirflickr

# image extension
# determines which images in IM_DIR will be included in current session
EXTENSION jpg

# output directory for tag files
OUT_DIR $IM_DIR\tags

# session directory
# contains the following files: tags.txt, current_image.txt and 
# IM_LIST (see below)
SESSION_DIR $IM_DIR\tagsession

# thumbnail directory
THUMBS_DIR $IM_DIR\thumbnails

# image list file 
# this file will be generated automatically, based on IM_DIR and EXTENSION, 
# whenever this file does not exist in SESSION_DIR; subsequently this file
# is used as long as it is not deleted! 
IM_LIST_FNAME image_list.txt

# subset file
# when left empty full image_list is used
# to select a subset, specify a tag file from OUT_DIR, e.g.
# SUBSET_FNAME sea.txt
SUBSET_FNAME

# grid mode view
N_ROWS 4
N_COLS 8

# size of thumbnails in pixels
THUMB_SIZE 160

# autoforward delay time (in ms)
DELAY 7000

###########################################################################
#
# TWEAKING 
# (probably best to leave the following settings alone!)
#
###########################################################################

# filename for tags file (manually generated by user)
# should be in SESSION_DIR
TAGS_FNAME tags.txt

# filename keeping track of current image
# generated automatically, but can be deleted to start at first image
POSITION_FNAME current_image.txt

# position and size of log console window
# in fractions of screen size
LOG_OFFSET_X 0.3
LOG_OFFSET_Y 0.65
LOG_WIDTH    0.6

# number of text lines in log console
LOG_LINES 15

# two settings for column printing in the console
N_TAG_COLUMNS 3
COLUMN_WIDTH  16

# used in TAGS_FNAME to get next image automatically after key press
# ignored in GRID view mode
AUTONEXT_CHAR +

# offset for top left corner of view window, in pixels 
VIEW_OFFSET 4

# image padding for single view mode
SINGLE_PADDING 30

# image padding for grid view mode
# determines width of highlighting rectangle
GRID_PADDING 3

# two settings for highlighting rectangle (grid view mode)
SELECT_BORDER_WIDTH 5
SELECT_BORDER_COLOR green

# extension of thumbnail files
THUMB_EXTENSION $EXTENSION

# suffix added to original image files to generate thumbnail filenames
# modify at your own risk...
THUMB_SUFFIX _t$THUMB_SIZE
